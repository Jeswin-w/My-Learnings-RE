## 1. Kotlin `when` Expression

```kotlin
fun main() {
    val day = 4

    val result = when (day) {
        1 -> "Monday"
        2 -> "Tuesday"
        3 -> "Wednesday"
        4 -> "Thursday"
        5 -> "Friday"
        6 -> "Saturday"
        7 -> "Sunday"
        else -> "Invalid day."
    }

    println(result)
}
```

## 2. Kotlin Arrays

### Create and Access Array
```kotlin
fun main() {
    val cars = arrayOf("Volvo", "BMW", "Ford", "Mazda")
    println(cars[0])
}
```

### Modify Array Element
```kotlin
fun main() {
    val cars = arrayOf("Volvo", "BMW", "Ford", "Mazda")
    cars[0] = "Opel"
    println(cars[0])
}
```

### Array Size
```kotlin
fun main() {
    val cars = arrayOf("Volvo", "BMW", "Ford", "Mazda")
    println(cars.size)
}
```

### Check Element Exists
```kotlin
fun main() {
    val cars = arrayOf("Volvo", "BMW", "Ford", "Mazda")

    if ("Volvo" in cars) {
        println("It exists!")
    }
}
```

### Loop Through Array
```kotlin
fun main() {
    val cars = arrayOf("Volvo", "BMW", "Ford", "Mazda")

    for (car in cars) {
        println(car)
    }
}
```

## 3. Kotlin Ranges

### Numeric Range
```kotlin
fun main() {
    for (i in 5..15) {
        println(i)
    }
}
```

### Character Range
```kotlin
fun main() {
    for (c in 'a'..'x') {
        println(c)
    }
}
```

### Check Value in Range
```kotlin
fun main() {
    val value = 8
    if (value in 5..15) {
        println("Value is in range")
    }
}
```

### break Example
```kotlin
fun main() {
    for (i in 5..15) {
        if (i == 10) break
        println(i)
    }
}
```

### continue Example
```kotlin
fun main() {
    for (i in 5..15) {
        if (i == 10) continue
        println(i)
    }
}
```
